--- FILE 1: public/robots.txt ---
Create public/robots.txt with EXACTLY this content (including the blank line before Sitemap:):

```
User-agent: *
Allow: /

Sitemap: https://redbirdlawnservice.com/sitemap-index.xml
```

CRITICAL: Use `sitemap-index.xml` not `sitemap.xml`. The @astrojs/sitemap integration generates
sitemap-index.xml as the entry point. Using sitemap.xml means Google cannot find it.

CLOUDFLARE WARNING (from research): After deploy, verify the served robots.txt matches this file exactly.
Cloudflare's "Instruct AI bot traffic with robots.txt" feature can silently prepend Disallow directives
for AI crawlers. Check Cloudflare Dashboard > Security > Bots and confirm that feature is OFF.
This check happens at deploy time, not now — just create the file.

--- FILE 2: public/og-image.png ---
Create a minimal 1200x630 placeholder PNG using Node.js canvas or a simple approach.

The simplest correct approach: use the Bash tool to run a Node.js one-liner that creates
a valid 1200x630 solid-color PNG using only the built-in `Buffer` and PNG binary format.

Run this Node.js script to generate the placeholder PNG:

```javascript
// Minimal PNG generator — no npm packages required
// Creates a solid #1a472a (dark green) 1200x630 PNG
const fs = require('fs');

function createMinimalPng(width, height, r, g, b) {
  // PNG header
  const header = Buffer.from([137, 80, 78, 71, 13, 10, 26, 10]);

  function crc32(buf) {
    let crc = -1;
    const table = [];
    for (let i = 0; i < 256; i++) {
      let c = i;
      for (let j = 0; j < 8; j++) c = (c & 1) ? (0xEDB88320 ^ (c >>> 1)) : (c >>> 1);
      table[i] = c;
    }
    for (const byte of buf) crc = table[(crc ^ byte) & 0xFF] ^ (crc >>> 8);
    return (crc ^ -1) >>> 0;
  }

  function chunk(type, data) {
    const len = Buffer.alloc(4);
    len.writeUInt32BE(data.length);
    const typeAndData = Buffer.concat([Buffer.from(type), data]);
    const crcBuf = Buffer.alloc(4);
    crcBuf.writeUInt32BE(crc32(typeAndData));
    return Buffer.concat([len, typeAndData, crcBuf]);
  }

  // IHDR chunk
  const ihdr = Buffer.alloc(13);
  ihdr.writeUInt32BE(width, 0);
  ihdr.writeUInt32BE(height, 4);
  ihdr[8] = 8;  // bit depth
  ihdr[9] = 2;  // color type: RGB
  ihdr[10] = 0; ihdr[11] = 0; ihdr[12] = 0;

  // IDAT chunk: uncompressed scanlines
  const zlib = require('zlib');
  const raw = Buffer.alloc(height * (1 + width * 3));
  for (let y = 0; y < height; y++) {
    const row = y * (1 + width * 3);
    raw[row] = 0; // filter type: None
    for (let x = 0; x < width; x++) {
      raw[row + 1 + x * 3] = r;
      raw[row + 2 + x * 3] = g;
      raw[row + 3 + x * 3] = b;
    }
  }
  const compressed = zlib.deflateSync(raw);
  const idat = chunk('IDAT', compressed);

  const result = Buffer.concat([
    header,
    chunk('IHDR', ihdr),
    idat,
    chunk('IEND', Buffer.alloc(0))
  ]);
  return result;
}

// Redbird green: #1a472a → RGB(26, 71, 42)
const png = createMinimalPng(1200, 630, 26, 71, 42);
fs.writeFileSync('public/og-image.png', png);
console.log('Created public/og-image.png — 1200x630 placeholder');
```

Save as a temporary script (e.g., /tmp/gen-og.js), run with `node /tmp/gen-og.js` from the project
root, then delete the temp script. Verify public/og-image.png exists and is > 0 bytes.

If for any reason the script approach fails, create the smallest valid PNG possible via any means
