**Add buildArticleSchema() to src/lib/schema.ts** — append AFTER the existing buildLocalBusinessSchema function. Do NOT modify the existing function:

```typescript
export interface ArticleSchemaProps {
  title: string;
  excerpt: string;
  slug: string;
  datePublished: Date;
  dateModified?: Date;
  authorName?: string;
}

export function buildArticleSchema(props: ArticleSchemaProps) {
  const { title, excerpt, slug, datePublished, dateModified, authorName = 'Alberto Murillo' } = props;
  return {
    "@context": "https://schema.org",
    "@type": "Article",
    "headline": title,
    "description": excerpt,
    "url": `${SITE_URL}/blog/${slug}`,
    "datePublished": datePublished.toISOString(),
    "dateModified": (dateModified ?? datePublished).toISOString(),
    "author": {
      "@type": "Person",
      "name": authorName,
    },
    "publisher": {
      "@type": "Organization",
      "name": NAP.name,
      "url": SITE_URL,
    },
    "mainEntityOfPage": {
      "@type": "WebPage",
      "@id": `${SITE_URL}/blog/${slug}`,
    },
  } as const;
}
```

```astro
// src/pages/blog/[...slug].astro
import Layout from '../../layouts/Layout.astro';
import { getCollection } from 'astro:content';
import { buildArticleSchema } from '../../lib/schema';
import { NAP } from '../../lib/constants';

export async function getStaticPaths() {
  const posts = await getCollection('blog');
  return posts.map((entry) => ({
    params: { slug: entry.slug },
    props: { entry },
  }));
}

const { entry } = Astro.props;
const { Content } = await entry.render();

const articleSchema = buildArticleSchema({
  title: entry.data.title,
  excerpt: entry.data.excerpt,
  slug: entry.slug,
  datePublished: entry.data.date,
  authorName: entry.data.author,
});

const ogTitle = `${entry.data.title} | Redbird Lawn Care`;
---

<Layout title={ogTitle} description={entry.data.excerpt}>
  <Fragment slot="head">
    <script type="application/ld+json" set:html={JSON.stringify(articleSchema)} />
  </Fragment>

  <article class="max-w-2xl mx-auto px-4 py-16">
    <!-- Breadcrumb -->
    <nav class="text-sm text-gray-400 dark:text-gray-500 mb-8" aria-label="Breadcrumb">
      <a href="/" class="hover:text-forest-green dark:hover:text-light-green transition-colors">Home</a>
```
